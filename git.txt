git init
git add filename 
git commit -m "new file" //多次add一次comiit并写信息
git status //查看仓库状态
git diff
git log [--pretty=oneline]
git reset --hard HEAD~1   //后面数字可选
git reset --hard 1094a      //1094a为版本号
git reflog   //记录之前的命令，方便恢复
git checkout  -- readme.txt  //将readme.txt恢复到最近一次commit或add状态
git rm file  //从版本库删除文件，同样需要commit，错删在没提交前同样使用checkout恢复
